Today:
Data visualization

Graphics with ggplot2

Four steps to visualization

1. Call ggplot()

ggplot()

2. Define data

ggplot(data = mpg)

3. Define aesthetic mappings

ggplot(data = mpg, aes(x = displ, y = hwy))

4. Add geometric object(s)

ggplot(data = mpg, aes(x = displ, y = hwy)) + 
  geom_point()

Some examples

Histograms

ggplot(data = mpg, aes(x = hwy)) + 
  geom_histogram()

Frequency polygons

ggplot(data = mpg, aes(x = hwy)) + 
  geom_freqpoly()

Densityplots

ggplot(data = mpg, aes(x = hwy)) + 
  geom_density()

Scatterplots

ggplot(data = mpg, aes(x = displ, y = hwy)) +

Lineplots

Maps

ggplot(map_data("world"), aes(long, lat, group = group)) +
  geom_polygon()

dino <- datasaurus_dozen %>% filter(dataset == "dino")

Combining plots

Histogramm

ggplot(data = mpg, aes(x = hwy)) + 
  geom_histogram()

Add a frequency polygon

ggplot(data = mpg, aes(x = hwy)) + 
  geom_histogram() +
  geom_freqpoly()

Scatterplot

ggplot(data = mpg, aes(x = displ, y = hwy)) + 
  geom_point()

Add a regression line

ggplot(data = mpg, aes(x = displ, y = hwy)) + 
  geom_point() +
  geom_smooth(method = "lm", se = FALSE)

Add a second regression line and conficence interval

ggplot(data = mpg, aes(x = displ, y = hwy)) + 
  geom_point() +
  geom_smooth(method = "lm", se = FALSE) +
  geom_smooth(method = "loess", se = TRUE)

Violin plot

ggplot(data = mpg, aes(x = drv, y = hwy)) + 
  geom_violin()

Add a box plot

ggplot(data = mpg, aes(x = drv, y = hwy)) + 
  geom_violin() +
  geom_boxplot(width = 0.1)

Edit your axis

Axis limits

ggplot(data = mpg, aes(x = displ, y = hwy)) + 
  geom_point() +
  scale_x_continuous(limits = c(0, 10))

Axis labels

ggplot(data = mpg, aes(x = displ, y = hwy)) + 
  geom_point() +
  labs(x = "Engine displacement", y = "Highway miles per gallon")

Axis scales

ggplot(data = mpg, aes(x = displ, y = hwy)) + 
  geom_point() +
  scale_x_continuous(limits = c(0, 10), breaks = c(0:10))

Other labels

Title

ggplot(data = mpg, aes(x = displ, y = hwy)) + 
  geom_point() +
  labs(title = "My Title")

Legend

ggplot(data = mpg, aes(x = displ, y = hwy, color = class)) + 
  geom_point() +
  labs(color = "My Legend")
